class MyNum
{
	int num;
public:
	explicit MyNum(int num)
	 {
		 this->num=num;
		 cout<<"in constr"<<num<<endl;
	 }
	 void disp()
	 {
		 cout<<num<<endl;
	 }
	 operator int()
	 {
		 return num;
	 }
	 void operator=(int k)
	{
		this->num=k;
		cout<<endl<<"in =operator function\n";
	}

};
void main()
{
     MyNum m1(20);					
	 m1=100; //Do not use Constructor for this purpose	
	 m1.disp();
	int k;						
	k=m1;							
	cout<<endl<<k<<endl;
}
